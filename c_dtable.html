示例代码

dk.ready(function(){
toolBar = [funcBtnType.query,funcBtnType['new'],funcBtnType.edit,funcBtnType.copy,funcBtnType['delete'],funcBtnType['refresh'],funcBtnType.save,funcBtnType['return']];
var contextArticleView = {
	name: '查看文章',
	icon: 'query.png',
	func: function(){
		if(d.getSelectedRowCount() != 1){
			return;
		}else{
			window.open('/article/view?aid=' + d.getGuidsString());
		}
	}
}; 
var contextMenuItems = ['context_new','context_edit','context_split', 'context_delete', contextArticleView];
var d = new dkGrid('dtable',toolBar,contextMenuItems);
